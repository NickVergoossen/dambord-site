                        #include <FastLED.h>
                        #define LED_PIN 2
                        #define NUM_LEDS 60
                        
                        CRGB leds[NUM_LEDS];
                        
                        int dirDown =  7;
                        int dirUp = 6;
                        int dirRight = 5;
                        int dirLeft = 4;
                        
                        void setup(){
                          pinMode(dirDown, INPUT);
                          pinMode(dirUp, INPUT);
                          pinMode(dirLeft, INPUT);
                          pinMode(dirRight, INPUT);
                        
                          Serial.begin(9600);
                        
                          FastLED.addLeds<WS2812, LED_PIN, GRB>(leds, NUM_LEDS);
                          FastLED.setMaxPowerInVoltsAndMilliamps(5, 500);
                          FastLED.clear();
                          FastLED.show();
                        }
                        
                        void loop(){
                        
                          if (digitalRead(dirDown) == LOW){
                            for (int i=0; i<NUM_LEDS; i++ )
                            {
                            leds[i] = CRGB(255, 0, 0 );
                            FastLED.setBrightness(2*i);
                            FastLED.show();
                            delay (20);
                            } 
                          }
                          else if (digitalRead(dirUp) == LOW){
                            for (int i=0; i<NUM_LEDS; i++ )
                            {
                            leds[i] = CRGB(0, 255, 255 );
                            FastLED.setBrightness(2*i);
                            FastLED.show();
                            delay (20);
                            } 
                          }
                          else if (digitalRead(dirLeft) == LOW){
                            for (int i=0; i<NUM_LEDS; i++ )
                            {
                            leds[i] = CRGB(255, 255, 255 );
                            FastLED.setBrightness(2*i);
                            FastLED.show();
                            delay (20);
                            } 
                          }
                          else if(digitalRead(dirRight) == LOW){
                            for (int i=0; i<NUM_LEDS; i++ )
                            {
                            leds[i] = CRGB(0, 255, 0 );
                            FastLED.setBrightness(2*i);
                            FastLED.show();
                            delay (20);
                            } 
                          }
                          else{
                            Serial.println("STOP");
                          }
                          delay(100);
                        }
